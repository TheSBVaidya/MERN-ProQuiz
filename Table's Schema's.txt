CREATE DATABASE IF NOT EXISTS exam_db; USE exam_db;

CREATE TABLE IF NOT EXISTS user ( id INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(255) NOT NULL, email VARCHAR(255) UNIQUE NOT NULL, password
VARCHAR(255) );

CREATE TABLE IF NOT EXISTS quiz ( id INT PRIMARY KEY, question TEXT, a
TEXT, b TEXT, c TEXT, d TEXT, correct_answer CHAR(10) );

CREATE TABLE IF NOT EXISTS attempts ( attempt_id INT AUTO_INCREMENT
PRIMARY KEY, user_id INT NOT NULL, created_at TIMESTAMP DEFAULT
CURRENT_TIMESTAMP );

CREATE TABLE IF NOT EXISTS attempt_answers ( id INT AUTO_INCREMENT
PRIMARY KEY, attempt_id INT NOT NULL, quiz_id INT NOT NULL, answer TEXT
NOT NULL );
